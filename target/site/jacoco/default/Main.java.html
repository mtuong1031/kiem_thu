public class Main {
    public static void main(String[] args) {
        LoginService loginService = new LoginService();
        
        // Test cases
        System.out.println("=== Testing Login Service ===");
        
        // Valid login
        System.out.println("Valid login: " + loginService.login("user123", "Password@123"));
        
        // Invalid username (too short)
        System.out.println("Invalid username (too short): " + loginService.login("user", "Password@123"));
        
        // Invalid password (too short)
        System.out.println("Invalid password (too short): " + loginService.login("user123", "Pass@1"));
        
        // Wrong credentials
        System.out.println("Wrong credentials: " + loginService.login("user123", "WrongPassword@123"));
        System.out.println("Wrong credentials: " + loginService.login("user123", "WrongPassword@124"));
        System.out.println("Wrong credentials: " + loginService.login("user123", "WrongPassword@125"));
        
        // Should be locked now
        System.out.println("After 3 failed attempts: " + loginService.login("user123", "Password@123"));
    }
}
